48void watchdog_start(void){
49 critical_enter();
50 wdt_reset();
51 MCUSR &= ~(1 << WDRF);
52 WDTCSR = (1 << WDCE) | (1 << WDE);
53 set_wdt_prescaler(WDT_PRESCALER_DIV128K);
54 critical_exit();
55}

57void watchdog_hold(void){
58 critical_enter();
59 wdt_reset();
60 MCUSR &= ~(1 << WDRF);
61 WDTCSR |= (1 << WDCE) | (1 << WDE);
62 WDTCSR = 0x00;
63 critical_exit();
64}

66void set_wdt_prescaler(WDT_PRESCALER prescaler) {
67 uint8_t wdp_bits = 0;
68
69 switch (prescaler) {
70 case WDT_PRESCALER_DIV2K:
71 wdp_bits = 0b0000;
72 break;
73 case WDT_PRESCALER_DIV4K:
74 wdp_bits = 0b0001;
75 break;
76 case WDT_PRESCALER_DIV8K:
77 wdp_bits = 0b0010;
78 break;
79 case WDT_PRESCALER_DIV16K:
80 wdp_bits = 0b0011;
81 break;
82 case WDT_PRESCALER_DIV32K:
83 wdp_bits = 0b0100;
84 break;
85 case WDT_PRESCALER_DIV64K:
86 wdp_bits = 0b0101;
87 break;
88 case WDT_PRESCALER_DIV128K:
89 wdp_bits = 0b0110;
90 break;
91 case WDT_PRESCALER_DIV256K:
92 wdp_bits = 0b0111;
93 break;
94 case WDT_PRESCALER_DIV512K:
95 wdp_bits = 0b1000;
96 break;
97 case WDT_PRESCALER_DIV1024K:
98 wdp_bits = 0b1001;
99 break;
100 }
101
102 // Set the WDTCSR register with the computed wdp_bits and enable the watchdog
103 WDTCSR = (1<<WDIE) | (1 << WDE) | (wdp_bits & 0b0111) | (wdp_bits &0b1000 << 2);
104}

136void clock_set_prescaler(uint16_t ps){
137 uint8_t clkps = 0;
138 switch (ps) {
139 case MCLK_PRESCALER_DIV1:
140 clkps = 0b0000;
141 break;
142 case MCLK_PRESCALER_DIV2:
143 clkps = 0b0001;
144 break;
145 case MCLK_PRESCALER_DIV4:
146 clkps = 0b0010;
147 break;
148 case MCLK_PRESCALER_DIV8:
149 clkps = 0b0011;
150 break;
151 case MCLK_PRESCALER_DIV16:
152 clkps = 0b0100;
153 break;
154 case MCLK_PRESCALER_DIV32:
155 clkps = 0b0101;
156 break;
157 case MCLK_PRESCALER_DIV64:
158 clkps = 0b0110;
159 break;
160 case MCLK_PRESCALER_DIV128:
161 clkps = 0b0111;
162 break;
163 case MCLK_PRESCALER_DIV256:
164 clkps = 0b1000;
165 break;
166 default:
167 die();
168 break;
169 }
170 critical_enter();
171 CLKPR = (1<<CLKPCE);
172 CLKPR = clkps & 0x0F;
173 critical_exit();
174}
