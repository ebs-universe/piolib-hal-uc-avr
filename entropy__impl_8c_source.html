46static inline void _entropy_state_init(void){
47 // If the size is not set, we assume we've never initialized the
48 // buffer. This has the implicit and possibly incorrect assumption
49 // that RAM will be reset to 0s on processor reset.
50 entropy_state.accumulating = 1;
51 if (!entropy_buf._size) {
52 bytebuf_vInit(entropy_buffer, uC_ENTROPY_POOL_SIZE, &entropy_buf);
53 }
54 if (!entropy_source._size) {
55 bytebuf_vInit(entropy_source_buffer, uC_ENTROPY_SOURCE_BUFFER_SIZE, &entropy_source);
56 }
57 entropy_state.entropy = &entropy_buf;
58 entropy_state.source = &entropy_source;
59}
60
61static inline void _entropy_state_deinit(void){
62 // If the size is not set, we assume we've never initialized the
63 // buffer. This has the implicit and possibly incorrect assumption
64 // that RAM will be reset to 0s on processor reset.
65 entropy_state.accumulating = 0;
66}

92void entropy_init(void){
93 // Note that watchdog resets are disabled in this condition.
94 // Further, if the application services the watchdog, the
95 // randomness generated will be a convolution of whatever is
96 // happening here and the loop that's controlling the watchdog
97 // service, and myst separately be validated.
98 entropy_backup = malloc(sizeof(entropy_backup_t));
99 critical_enter();
100 wdt_reset();
101 MCUSR = 0;
102 WDTCSR = (1 << WDCE) | (1 << WDE);
103 WDTCSR = (1 << WDIE);
104 critical_exit();
105 // TODO Save timer state. Also disable interrupts if set.
106 timer_init(uC_ENTROPY_TIMER);
107 timer_set_prescaler(
108 uC_ENTROPY_TIMER,
109 TIMER_PRESCALER_DIV1
110 );
111 timer_set_mode(
112 uC_ENTROPY_TIMER,
113 TIMER_MODE_COUNTING
114 );
115 _entropy_state_init();
116}
117
118void entropy_deinit(void){
119 _entropy_state_deinit();
120 // TODO Restore timer state instead.
121 timer_set_mode(
122 uC_ENTROPY_TIMER,
123 TIMER_MODE_STOPPED
124 );
125 critical_enter();
126 wdt_reset();
127 MCUSR = 0;
128 WDTCSR = (1 << WDCE) | (1 << WDE);
129 WDTCSR = 0x00;
130 // TODO This causes a reset loop. Work it out later.
131 // WDTCSR = entropy_backup->wdtcsr_bak;
132 critical_exit();
133 free(entropy_backup);
134}
