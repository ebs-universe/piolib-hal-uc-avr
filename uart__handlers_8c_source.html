55 static inline void uart0_handler_rx(void){
56 uint8_t rval;
57 rval = HWREG8(uC_UART0_BASE + OFS_UART_UDR0);
58 if (bytebuf_cGetFree(&uart0_rxbuf)){
59 bytebuf_cPushByte(&uart0_rxbuf, rval, 0x00);
60 }
61 else{
62 (*uart0_if.state->overrun_counter)++;
63 }
64 }
65
66 static inline void uart0_handler_tx(void){
67 uint8_t waiting = bytebuf_cPopulation(&uart0_txbuf);
68 if (waiting <= 1){
69 // Disable the UDRE interrupt
70 HWREG8(uC_UART0_BASE + OFS_UART_UCSR0B) &= ~(1 << UDRIE0);
71 uart0_if.state->triggered = 0;
72 }
73 if (waiting) {
74 HWREG8(uC_UART0_BASE + OFS_UART_UDR0) = bytebuf_cPopByte(&uart0_txbuf);
75 }
76 }

78 ISR(USART_RX_vect) {
